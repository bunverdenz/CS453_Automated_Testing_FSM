  function fillvoicelist(){
    var sel = document.getElementById('voicelist');
    // create new option element
    var opt;

// add opt to end of select box (sel)
 
    for (var i = 0; i < 74; i++) {
        opt = document.createElement('option');
        opt.appendChild( document.createTextNode(voicesinfo[i][1] + '(' +voicesinfo[i][3]+ ')') );
        opt.value = "335200" + voicesinfo[i][0].substring(2, 4);
        sel.appendChild(opt); 
    }
    
   
  }
